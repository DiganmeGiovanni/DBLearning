
Crear usuario:
CREATE USER DGIO IDENTIFIED BY SOMEPASS;

Asignar privilegios y permisos:
GRANT CONNECT, RESOURCE, CREATE TABLE TO DGIO;

Asignar un tablespace a un usuario
ALTER USER DGIO DEFAULT TABLESPACE SOMETBSNAME;

Consultar el tablespace asignado para un usuario:
SELECT USERNAME, DEFAULT_TABLESPACE FROM DBA_USERS;
INSERT INTO CARS_BYLIST VALUES(2001, 'FORD');

Tablespace temporal:
CREATE TEMPORARY TABLESPACE TEMPTBS TEMPFILE '/practicas/datafiles/temptbs01.dbf' SIZE 10M
ALTER DATABASE DEFAULT TEMPORARY TABLESPACE TEMPTBS;
SELECT PROPERTY_VALUE FROM DATABASE_PROPERTIES WHERE PROPERTY_NAME= 'DEFAULT_TEMP_TABLESPACE';
